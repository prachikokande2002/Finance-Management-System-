---------------------------expense.py â€” Expense Class (OOP)----------
class Expense:
    def __init__(self, amount, category, date, description):
        self.amount = float(amount)
        self.category = category
        self.date = date
        self.description = description

    def to_list(self):
        return [self.amount, self.category, self.date, self.description]

    @staticmethod
    def from_list(data):
        return Expense(
            amount=float(data[0]),
            category=data[1],
            date=data[2],
            description=data[3]
        )


---------------------file_manager.py â€” CSV Persistence + Backup------------

import csv
import shutil
from expense import Expense

DATA_FILE = "expenses.csv"
BACKUP_FILE = "expenses_backup.csv"

def save_expense(expense):
    with open(DATA_FILE, "a", newline="") as file:
        writer = csv.writer(file)
        writer.writerow(expense.to_list())

def load_expenses():
    expenses = []
    try:
        with open(DATA_FILE, "r") as file:
            reader = csv.reader(file)
            for row in reader:
                expenses.append(Expense.from_list(row))
    except FileNotFoundError:
        pass
    return expenses

def backup_data():
    try:
        shutil.copy(DATA_FILE, BACKUP_FILE)
    except FileNotFoundError:
        print("No data file found to backup.")

def restore_data():
    try:
        shutil.copy(BACKUP_FILE, DATA_FILE)
    except FileNotFoundError:
        print("No backup file found.")


-------------------------reports.py â€” Reports & Analysis-------------------

from collections import defaultdict

def generate_report(expenses):
    total = sum(exp.amount for exp in expenses)
    average = total / len(expenses) if expenses else 0

    category_totals = defaultdict(float)
    for exp in expenses:
        category_totals[exp.category] += exp.amount

    return total, average, category_totals

def display_report(expenses):
    total, average, category_totals = generate_report(expenses)

    print("\nðŸ“Š EXPENSE REPORT")
    print(f"Total Expenses: ${total:.2f}")
    print(f"Average Expense: ${average:.2f}")

    print("\nCategory-wise Spending:")
    for category, amount in category_totals.items():
        print(f"  {category}: ${amount:.2f}")

------------------------utils.py â€” Validation & Formatting----------------


from datetime import datetime

def validate_amount(value):
    try:
        value = float(value)
        if value <= 0:
            raise ValueError
        return value
    except ValueError:
        raise ValueError("Amount must be a positive number.")

def validate_date(date_str):
    try:
        datetime.strptime(date_str, "%Y-%m-%d")
        return date_str
    except ValueError:
        raise ValueError("Date format must be YYYY-MM-DD.")

def validate_text(text):
    if not text.strip():
        raise ValueError("This field cannot be empty.")
    return text.strip()
